# 2-11. コンピューティング

AWSのコンピューティングサービス（EC2、Lambda、API Gatewayなど）の構築方法を説明します。

## 2-11-1. EC2インスタンス

### EC2インスタンスの基本作成

基本的なEC2インスタンスを作成します。

```hcl
resource "aws_instance" "web" {
  ami           = "ami-0c3fd0f5d33134a76"  # Amazon Linux 2
  instance_type = "t3.micro"
  
  tags = {
    Name = "web-server"
  }
}
```

### インスタンスタイプの選択

用途に応じてインスタンスタイプを選択します。

```hcl
# 汎用インスタンス
resource "aws_instance" "general" {
  instance_type = "t3.micro"  # 小規模なアプリケーション
  # instance_type = "t3.medium"  # 中規模なアプリケーション
  # instance_type = "t3.large"   # 大規模なアプリケーション
}

# コンピューティング最適化
resource "aws_instance" "compute" {
  instance_type = "c5.large"  # 高CPU性能が必要な場合
}

# メモリ最適化
resource "aws_instance" "memory" {
  instance_type = "r5.large"  # 大量のメモリが必要な場合
}
```

### AMIの選択

最新のAMIを動的に取得します。

```hcl
data "aws_ami" "amazon_linux" {
  most_recent = true
  owners      = ["amazon"]
  
  filter {
    name   = "name"
    values = ["amzn2-ami-hvm-*-x86_64-gp2"]
  }
}

resource "aws_instance" "web" {
  ami           = data.aws_ami.amazon_linux.id
  instance_type = "t3.micro"
}
```

### キーペアの設定

既存のキーペアを使用します。

```hcl
resource "aws_instance" "web" {
  ami           = data.aws_ami.amazon_linux.id
  instance_type = "t3.micro"
  key_name      = "my-key-pair"
}
```

### セキュリティグループの設定

セキュリティグループを設定します。

```hcl
resource "aws_security_group" "web" {
  name        = "web-sg"
  description = "Webサーバー用のセキュリティグループ"
  vpc_id      = aws_vpc.main.id
  
  ingress {
    from_port   = 80
    to_port     = 80
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }
  
  ingress {
    from_port   = 443
    to_port     = 443
    protocol    = "tcp"
    cidr_blocks = ["0.0.0.0/0"]
  }
  
  egress {
    from_port   = 0
    to_port     = 0
    protocol    = "-1"
    cidr_blocks = ["0.0.0.0/0"]
  }
}

resource "aws_instance" "web" {
  ami                    = data.aws_ami.amazon_linux.id
  instance_type          = "t3.micro"
  vpc_security_group_ids = [aws_security_group.web.id]
  subnet_id              = aws_subnet.public.id
}
```

### ユーザーデータの設定

インスタンス起動時にスクリプトを実行します。

```hcl
resource "aws_instance" "web" {
  ami           = data.aws_ami.amazon_linux.id
  instance_type = "t3.micro"
  
  user_data = <<-EOF
    #!/bin/bash
    yum update -y
    yum install -y httpd
    systemctl start httpd
    systemctl enable httpd
    echo "<h1>Hello from Terraform</h1>" > /var/www/html/index.html
  EOF
}
```

### IAMロールのアタッチ

IAMロールをアタッチします。

```hcl
resource "aws_iam_role" "ec2_role" {
  name = "ec2-role"
  
  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Action = "sts:AssumeRole"
      Effect = "Allow"
      Principal = {
        Service = "ec2.amazonaws.com"
      }
    }]
  })
}

resource "aws_iam_instance_profile" "ec2_profile" {
  name = "ec2-instance-profile"
  role = aws_iam_role.ec2_role.name
}

resource "aws_instance" "web" {
  ami                  = data.aws_ami.amazon_linux.id
  instance_type        = "t3.micro"
  iam_instance_profile = aws_iam_instance_profile.ec2_profile.name
}
```

### EBSボリュームのアタッチ

EBSボリュームをアタッチします。

```hcl
resource "aws_instance" "web" {
  ami           = data.aws_ami.amazon_linux.id
  instance_type = "t3.micro"
  
  root_block_device {
    volume_type = "gp3"
    volume_size = 20
    encrypted   = true
  }
  
  ebs_block_device {
    device_name = "/dev/sdf"
    volume_type = "gp3"
    volume_size = 100
    encrypted   = true
  }
}
```

### タグの設定

インスタンスにタグを設定します。

```hcl

  ami           = data.aws_ami.amazon_linux.id

```

## 2-11-2. オートスケーリング

### 起動テンプレートの作成

オートスケーリング用の起動テンプレートを作成します。

```hcl
resource "aws_launch_template" "web" {
  name_prefix   = "web-"
  image_id      = data.aws_ami.amazon_linux.id
  instance_type = "t3.micro"
  
  vpc_security_group_ids = [aws_security_group.web.id]
  
  user_data = base64encode(<<-EOF
    #!/bin/bash
    yum update -y
    yum install -y httpd
    systemctl start httpd
    systemctl enable httpd
  EOF
  )
  
  tag_specifications {
    resource_type = "instance"
    tags = {
      Name = "web-server"
    }
  }
}
```

### オートスケーリンググループの作成

オートスケーリンググループを作成します。

```hcl
resource "aws_autoscaling_group" "web" {
  name                = "web-asg"
  vpc_zone_identifier = [aws_subnet.public.id]
  target_group_arns   = [aws_lb_target_group.web.arn]
  health_check_type   = "ELB"
  min_size            = 1
  max_size            = 5
  desired_capacity    = 2
  
  launch_template {
    id      = aws_launch_template.web.id
    version = "$Latest"
  }
  
  tag {
    key                 = "Name"
    value               = "web-asg-instance"
    propagate_at_launch = true
  }
}
```

### スケーリングポリシーの設定

スケーリングポリシーを設定します。

```hcl
resource "aws_autoscaling_policy" "scale_up" {
  name                   = "scale-up"
  autoscaling_group_name = aws_autoscaling_group.web.name
  adjustment_type        = "ChangeInCapacity"
  scaling_adjustment     = 1
  cooldown               = 300
}

resource "aws_autoscaling_policy" "scale_down" {
  name                   = "scale-down"
  autoscaling_group_name = aws_autoscaling_group.web.name
  adjustment_type        = "ChangeInCapacity"
  scaling_adjustment     = -1
  cooldown               = 300
}
```

### ヘルスチェックの設定

ヘルスチェックを設定します。

```hcl
resource "aws_autoscaling_group" "web" {
  health_check_type         = "ELB"
  health_check_grace_period = 300
  
  # ヘルスチェックが失敗したインスタンスを置き換え
  force_delete = true
}
```

### ライフサイクルフックの設定

ライフサイクルフックを設定します。

```hcl
resource "aws_autoscaling_lifecycle_hook" "web" {
  name                   = "web-lifecycle-hook"
  autoscaling_group_name = aws_autoscaling_group.web.name
  default_result         = "CONTINUE"
  heartbeat_timeout      = 2000
  lifecycle_transition   = "autoscaling:EC2_INSTANCE_LAUNCHING"
}
```

## 2-11-3. Lambda関数

### Lambda関数の基本作成

基本的なLambda関数を作成します。

```hcl
resource "aws_lambda_function" "example" {
  filename      = "lambda_function.zip"
  function_name = "example-function"
  role          = aws_iam_role.lambda_role.arn
  handler       = "index.handler"
  runtime       = "python3.11"
}
```

### ランタイムの選択

使用するランタイムを選択します。

```hcl
# Python
resource "aws_lambda_function" "python" {
  runtime = "python3.11"
}

# Node.js
resource "aws_lambda_function" "nodejs" {
  runtime = "nodejs20.x"
}

# Java
resource "aws_lambda_function" "java" {
  runtime = "java17"
}

# Go
resource "aws_lambda_function" "go" {
  runtime = "provided.al2"
}
```

### ハンドラーの設定

ハンドラー関数を設定します。

```hcl
# Pythonの場合
resource "aws_lambda_function" "example" {
  handler = "lambda_function.lambda_handler"  # ファイル名.関数名
  runtime = "python3.11"
}

# Node.jsの場合
resource "aws_lambda_function" "example" {
  handler = "index.handler"  # ファイル名.関数名
  runtime = "nodejs20.x"
}
```

### 環境変数の設定

環境変数を設定します。

```hcl
resource "aws_lambda_function" "example" {
  function_name = "example-function"
  handler       = "index.handler"
  runtime       = "nodejs20.x"
  
  environment {
    variables = {
      ENVIRONMENT = "production"
      API_KEY     = var.api_key
      DB_HOST     = aws_rds_cluster.example.endpoint
    }
  }
}
```

### メモリとタイムアウトの設定

メモリとタイムアウトを設定します。

```hcl
resource "aws_lambda_function" "example" {
  function_name = "example-function"
  handler       = "index.handler"
  runtime       = "nodejs20.x"
  memory_size   = 512   # MB（128〜10240）
  timeout       = 30    # 秒（最大900秒）
}
```

### IAMロールの設定

Lambda関数用のIAMロールを設定します。

```hcl
resource "aws_iam_role" "lambda_role" {
  name = "lambda-execution-role"
  
  assume_role_policy = jsonencode({
    Version = "2012-10-17"
    Statement = [{
      Action = "sts:AssumeRole"
      Effect = "Allow"
      Principal = {
        Service = "lambda.amazonaws.com"
      }
    }]
  })
}

resource "aws_iam_role_policy_attachment" "lambda_basic" {
  role       = aws_iam_role.lambda_role.name
  policy_arn = "arn:aws:iam::aws:policy/service-role/AWSLambdaBasicExecutionRole"
}
```

### VPC設定

Lambda関数をVPC内で実行します。

```hcl
resource "aws_lambda_function" "example" {
  function_name = "example-function"
  handler       = "index.handler"
  runtime       = "nodejs20.x"
  
  vpc_config {
    subnet_ids         = [aws_subnet.private.id]
    security_group_ids = [aws_security_group.lambda.id]
  }
}
```

### デッドレターキュー（DLQ）の設定

エラー処理用のDLQを設定します。

```hcl
resource "aws_sqs_queue" "dlq" {
  name = "lambda-dlq"
}

resource "aws_lambda_function" "example" {
  function_name = "example-function"
  handler       = "index.handler"
  runtime       = "nodejs20.x"
  
  dead_letter_config {
    target_arn = aws_sqs_queue.dlq.arn
  }
}
```

### イベントソースマッピング

SQSやDynamoDBストリームなどのイベントソースを設定します。

```hcl
resource "aws_lambda_event_source_mapping" "sqs" {
  event_source_arn = aws_sqs_queue.example.arn
  function_name    = aws_lambda_function.example.arn
  batch_size       = 10
}
```

### Lambdaレイヤーの作成

Lambdaレイヤーを作成します。

```hcl
resource "aws_lambda_layer_version" "example" {
  filename            = "layer.zip"
  layer_name          = "example-layer"
  compatible_runtimes = ["python3.11"]
}

resource "aws_lambda_function" "example" {
  function_name = "example-function"
  handler       = "index.handler"
  runtime       = "python3.11"
  
  layers = [aws_lambda_layer_version.example.arn]
}
```

## 2-11-4. API Gateway

### REST APIの作成

REST APIを作成します。

```hcl
resource "aws_api_gateway_rest_api" "example" {
  name        = "example-api"
  description = "Example API"
}
```

### リソースとメソッドの定義

リソースとHTTPメソッドを定義します。

```hcl
resource "aws_api_gateway_resource" "users" {
  rest_api_id = aws_api_gateway_rest_api.example.id
  parent_id   = aws_api_gateway_rest_api.example.root_resource_id
  path_part   = "users"
}

resource "aws_api_gateway_method" "get_users" {
  rest_api_id   = aws_api_gateway_rest_api.example.id
  resource_id   = aws_api_gateway_resource.users.id
  http_method   = "GET"
  authorization = "NONE"
}
```

### 統合の設定（Lambda、HTTP、Mock）

#### Lambda統合

```hcl
resource "aws_api_gateway_integration" "lambda" {
  rest_api_id = aws_api_gateway_rest_api.example.id
  resource_id = aws_api_gateway_resource.users.id
  http_method = aws_api_gateway_method.get_users.http_method
  
  integration_http_method = "POST"
  type                    = "AWS_PROXY"
  uri                     = aws_lambda_function.example.invoke_arn
}
```

#### HTTP統合

```hcl
resource "aws_api_gateway_integration" "http" {
  rest_api_id = aws_api_gateway_rest_api.example.id
  resource_id = aws_api_gateway_resource.users.id
  http_method = aws_api_gateway_method.get_users.http_method
  
  type = "HTTP"
  uri  = "https://api.example.com/users"
}
```

#### Mock統合

```hcl
resource "aws_api_gateway_integration" "mock" {
  rest_api_id = aws_api_gateway_rest_api.example.id
  resource_id = aws_api_gateway_resource.users.id
  http_method = aws_api_gateway_method.get_users.http_method
  
  type = "MOCK"
  
  request_templates = {
    "application/json" = jsonencode({
      statusCode = 200
    })
  }
}
```

### デプロイステージの作成

APIをデプロイするステージを作成します。

```hcl
resource "aws_api_gateway_deployment" "example" {
  rest_api_id = aws_api_gateway_rest_api.example.id
  
  triggers = {
    redeployment = sha1(jsonencode([
      aws_api_gateway_resource.users.id,
      aws_api_gateway_method.get_users.id,
      aws_api_gateway_integration.lambda.id,
    ]))
  }
  
  lifecycle {
    create_before_destroy = true
  }
}

resource "aws_api_gateway_stage" "example" {
  deployment_id = aws_api_gateway_deployment.example.id
  rest_api_id   = aws_api_gateway_rest_api.example.id
  stage_name    = "prod"
}
```

### カスタムドメインの設定

カスタムドメインを設定します。

```hcl
resource "aws_api_gateway_domain_name" "example" {
  domain_name     = "api.example.com"
  certificate_arn = aws_acm_certificate.example.arn
}

resource "aws_api_gateway_base_path_mapping" "example" {
  api_id      = aws_api_gateway_rest_api.example.id
  stage_name  = aws_api_gateway_stage.example.stage_name
  domain_name = aws_api_gateway_domain_name.example.domain_name
}
```

### APIキーの設定

APIキーを作成して使用量プランに追加します。

```hcl
resource "aws_api_gateway_api_key" "example" {
  name = "example-api-key"
}

resource "aws_api_gateway_usage_plan" "example" {
  name = "example-usage-plan"
  
  api_stages {
    api_id = aws_api_gateway_rest_api.example.id
    stage  = aws_api_gateway_stage.example.stage_name
  }
}

resource "aws_api_gateway_usage_plan_key" "example" {
  key_id        = aws_api_gateway_api_key.example.id
  key_type      = "API_KEY"
  usage_plan_id = aws_api_gateway_usage_plan.example.id
}
```

### 使用量プランの設定

APIの使用量を制限するプランを設定します。

```hcl
resource "aws_api_gateway_usage_plan" "example" {
  name = "example-usage-plan"
  
  throttle_settings {
    burst_limit = 100
    rate_limit  = 50
  }
  
  quota_settings {
    limit  = 10000
    period = "DAY"
  }
}
```

### 認証の設定（Cognito、Lambda Authorizer）

#### Cognito認証

```hcl
resource "aws_api_gateway_authorizer" "cognito" {
  name                   = "cognito-authorizer"
  rest_api_id            = aws_api_gateway_rest_api.example.id
  type                   = "COGNITO_USER_POOLS"
  provider_arns          = [aws_cognito_user_pool.example.arn]
  identity_source        = "method.request.header.Authorization"
}

resource "aws_api_gateway_method" "get_users" {
  authorization = "COGNITO_USER_POOLS"
  authorizer_id = aws_api_gateway_authorizer.cognito.id
}
```

#### Lambda Authorizer

```hcl
resource "aws_api_gateway_authorizer" "lambda" {
  name                   = "lambda-authorizer"
  rest_api_id            = aws_api_gateway_rest_api.example.id
  authorizer_uri         = aws_lambda_function.authorizer.invoke_arn
  authorizer_credentials = aws_iam_role.api_gateway.arn
}

resource "aws_api_gateway_method" "get_users" {
  authorization = "CUSTOM"
  authorizer_id = aws_api_gateway_authorizer.lambda.id
}
```
